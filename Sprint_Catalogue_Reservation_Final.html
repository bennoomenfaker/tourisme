3.1.3 Table offer_items NOUVELLE

Variantes vendables d'une offre (ex : « Randonnée matin », « Randonnée journée complète »).

CREATE TABLE offer_items (
  id                       UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  offer_id                 UUID NOT NULL REFERENCES offers(id) ON DELETE CASCADE,
  title                    VARCHAR NOT NULL,
  description              TEXT,
  duration                 VARCHAR,
  max_participants         INT,
  global_stock             INT,             -- stock total (catalogue simple)
  cancellation_policy_type VARCHAR DEFAULT 'flexible',  -- flexible | moderate | strict
  details                  JSONB,           -- infos complémentaires (matériel, prérequis…)
  sort_order               INT DEFAULT 0,
  is_active                BOOLEAN DEFAULT true,
  created_at               TIMESTAMPTZ DEFAULT NOW(),
  updated_at               TIMESTAMPTZ DEFAULT NOW()
);

3.1.4 Table offer_item_prices NOUVELLE

Prix par catégorie de participant pour chaque item.

CREATE TABLE offer_item_prices (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  offer_item_id UUID NOT NULL REFERENCES offer_items(id) ON DELETE CASCADE,
  label         VARCHAR NOT NULL,          -- "Adulte", "Enfant (6-12 ans)", "Étudiant"
  price         DECIMAL(10,2) NOT NULL,
  description   TEXT,
  sort_order    INT DEFAULT 0,
  created_at    TIMESTAMPTZ DEFAULT NOW()
);

3.2 Catalogue COMPLEXE SPRINT 4

Logique : On ajoute la dimension calendaire. Des règles de disponibilité (ex : « tous les lundis à 9h ») → génèrent des sessions concrètes → les réservations bookent des sessions. La capacité est traquée par date via offer_item_capacity. Idéal pour : éco-tours, activités, ateliers, transferts.

3.2.1 Table offer_item_availability_rules NOUVELLE

Règles de disponibilité récurrentes (ex : lundi 9h-12h, capacité 15).

CREATE TABLE offer_item_availability_rules (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  offer_item_id UUID NOT NULL REFERENCES offer_items(id) ON DELETE CASCADE,
  day_of_week   INT NOT NULL CHECK (day_of_week BETWEEN 0 AND 6),  -- 0=Dim, 6=Sam
  start_time    TIME NOT NULL,
  end_time      TIME NOT NULL,
  capacity      INT NOT NULL,
  is_active     BOOLEAN DEFAULT true,
  valid_from    DATE,
  valid_until   DATE,
  created_at    TIMESTAMPTZ DEFAULT NOW()
);

3.2.2 Table offer_item_capacity NOUVELLE

Capacité journalière résiduelle pour un item.

CREATE TABLE offer_item_capacity (
  id             UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  offer_item_id  UUID NOT NULL REFERENCES offer_items(id) ON DELETE CASCADE,
  date           DATE NOT NULL,
  total_capacity INT NOT NULL,
  booked_count   INT DEFAULT 0,
  created_at     TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(offer_item_id, date)
);

3.2.3 Table sessions NOUVELLE

Créneaux concrets (générés depuis les règles ou créés manuellement).

CREATE TABLE sessions (
  id               UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  offer_item_id    UUID NOT NULL REFERENCES offer_items(id) ON DELETE CASCADE,
  date             DATE NOT NULL,
  start_time       TIME NOT NULL,
  end_time         TIME NOT NULL,
  max_participants INT NOT NULL,
  booked_count     INT DEFAULT 0,
  status           VARCHAR DEFAULT 'scheduled',  -- scheduled | in_progress | completed | cancelled
  notes            TEXT,
  created_at       TIMESTAMPTZ DEFAULT NOW(),
  updated_at       TIMESTAMPTZ DEFAULT NOW()
);

3.2.4 Table bookings NOUVELLE

Réservations effectuées par les voyageurs (cœur du système).

CREATE TABLE bookings (
  id                UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  booking_ref       VARCHAR UNIQUE NOT NULL,      -- référence lisible (ex: "BK-XXXXXX")
  traveler_id       UUID NOT NULL REFERENCES users(id),
  offer_id          UUID NOT NULL REFERENCES offers(id),
  offer_item_id     UUID REFERENCES offer_items(id),
  session_id        UUID REFERENCES sessions(id),
  status            VARCHAR DEFAULT 'pending',    -- pending | confirmed | cancelled | completed | refunded
  total_price       DECIMAL(10,2) NOT NULL,
  currency          VARCHAR DEFAULT 'XAF',
  special_requests  TEXT,
  confirmation_mode VARCHAR,                      -- automatic | manual
  cancelled_at      TIMESTAMPTZ,
  cancel_reason     TEXT,
  created_at        TIMESTAMPTZ DEFAULT NOW(),
  updated_at        TIMESTAMPTZ DEFAULT NOW()
);

3.2.5 Table booking_participants NOUVELLE

Participants individuels d'une réservation de groupe.

CREATE TABLE booking_participants (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  booking_id      UUID NOT NULL REFERENCES bookings(id) ON DELETE CASCADE,
  full_name       VARCHAR NOT NULL,
  age             INT,
  document_type   VARCHAR,             -- passport | id_card | none
  document_number VARCHAR,
  is_group_leader BOOLEAN DEFAULT false,
  created_at      TIMESTAMPTZ DEFAULT NOW()
);

3.3 Circuits SPRINT 5

Logique : Un circuit est un package multi-étapes (plusieurs jours). Il a ses propres options de départ (dates, prix, capacité). Les réservations de circuits pointent vers la table bookings via circuit_reservation_options.booking_id.

3.3.1 Table circuits NOUVELLE

CREATE TABLE circuits (
  id                UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  author_id         UUID NOT NULL,
  author_type       VARCHAR NOT NULL,         -- 'guide' | 'project_owner'
  project_id        UUID REFERENCES projects(id),
  title             VARCHAR NOT NULL,
  description       TEXT,
  duration          VARCHAR NOT NULL,          -- "3 jours / 2 nuits"
  region            VARCHAR,
  total_price       DECIMAL(10,2),
  currency          VARCHAR DEFAULT 'XAF',
  max_participants  INT,
  inclusions        TEXT,
  exclusions        TEXT,
  status            VARCHAR DEFAULT 'pending', -- pending | approved | rejected | archived
  rejection_reason  TEXT,
  created_at        TIMESTAMPTZ DEFAULT NOW(),
  updated_at        TIMESTAMPTZ DEFAULT NOW()
);

3.3.2 Table circuit_stops NOUVELLE

CREATE TABLE circuit_stops (
  id                 UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  circuit_id         UUID NOT NULL REFERENCES circuits(id) ON DELETE CASCADE,
  day_number         INT NOT NULL,
  stop_order         INT NOT NULL,
  title              VARCHAR NOT NULL,
  description        TEXT,
  latitude           DECIMAL(10,7),
  longitude          DECIMAL(10,7),
  duration           VARCHAR,                -- "2h", "1 journée"
  accommodation_type VARCHAR,                -- hotel | lodge | camping | homestay
  meal_included      VARCHAR,                -- breakfast | half_board | full_board | none
  created_at         TIMESTAMPTZ DEFAULT NOW()
);

3.3.3 Table circuit_reservation_options NOUVELLE

CREATE TABLE circuit_reservation_options (
  id                UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  circuit_id        UUID NOT NULL REFERENCES circuits(id) ON DELETE CASCADE,
  start_date        DATE NOT NULL,
  end_date          DATE NOT NULL,
  price_per_person  DECIMAL(10,2) NOT NULL,
  max_participants  INT NOT NULL,
  booked_count      INT DEFAULT 0,
  status            VARCHAR DEFAULT 'available',  -- available | full | cancelled
  booking_id        UUID REFERENCES bookings(id) ON DELETE SET NULL,
  created_at        TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(circuit_id, start_date, end_date)
);

3.4 Notifications SPRINT 5

Logique : Système de notification unifié. Chaque événement métier (réservation confirmée, offre approuvée, etc.) génère une ligne dans notifications pour l'utilisateur concerné. 
CREATE TABLE notifications (
  id         UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id    UUID NOT NULL REFERENCES users(id),
  type       VARCHAR NOT NULL,   -- booking_confirmed | booking_cancelled | booking_request |
                                 -- offer_approved | offer_rejected | new_message | circuit_available
  title      VARCHAR NOT NULL,
  body       TEXT,
  link       VARCHAR,            -- lien profond vers la page concernée
  is_read    BOOLEAN DEFAULT false,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_notifications_user_unread
  ON notifications(user_id, is_read, created_at DESC);
